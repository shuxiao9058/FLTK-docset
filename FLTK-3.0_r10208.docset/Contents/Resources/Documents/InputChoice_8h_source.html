62  class FLTK3_EXPORT InputChoice : public fltk3::Group {
63  // Private class to handle slightly 'special' behavior of menu button
64  class InputMenuButton : public fltk3::MenuButton {
65  void draw() {
66  draw_box(fltk3::UP_BOX, color());
67  fltk3::color(active_r() ? labelcolor() : fltk3::inactive(labelcolor()));
68  int xc = w()/2, yc=h()/2;
69  fltk3::polygon(xc-5,yc-3,xc+5,yc-3,xc,yc+3);
70  if (fltk3::focus() == this) draw_focus();
71  }
72  public:
73  InputMenuButton(int x,int y,int w,int h,const char*l=0) :
74  fltk3::MenuButton(x,y,w,h,l) { box(fltk3::UP_BOX); }
75  };

80  static void menu_cb(fltk3::Widget*, void *data) {
81  fltk3::InputChoice *o=(fltk3::InputChoice *)data;
82  fltk3::WidgetTracker wp(o);
83  const fltk3::MenuItem *item = o->menubutton()->mvalue();
84  if (item && item->flags & (fltk3::SUBMENU|fltk3::SUBMENU_POINTER)) return; // ignore submenus
85  if (!strcmp(o->inp_->value(), o->menu_->text()))
86  {
87  o->Widget::clear_changed();
88  if (o->when() & fltk3::WHEN_NOT_CHANGED)
89  o->do_callback();
90  }
91  else
92  {
93  o->inp_->value(o->menu_->text());
94  o->inp_->set_changed();
95  o->Widget::set_changed();
96  if (o->when() & (fltk3::WHEN_CHANGED|fltk3::WHEN_RELEASE))
97  o->do_callback();
98  }
99 
100  if (wp.deleted()) return;
101 
102  if (o->callback() != default_callback)
103  {
104  o->Widget::clear_changed();
105  o->inp_->clear_changed();
106  }
107  }
108 
109  static void inp_cb(fltk3::Widget*, void *data) {
110  fltk3::InputChoice *o=(fltk3::InputChoice *)data;
111  fltk3::WidgetTracker wp(o);
112  if (o->inp_->changed()) {
113  o->Widget::set_changed();
114  if (o->when() & (fltk3::WHEN_CHANGED|fltk3::WHEN_RELEASE))
115  o->do_callback();
116  } else {
117  o->Widget::clear_changed();
118  if (o->when() & fltk3::WHEN_NOT_CHANGED)
119  o->do_callback();
120  }
121 
122  if (wp.deleted()) return;
123 
124  if (o->callback() != default_callback)
125  o->Widget::clear_changed();
126  }

128  // Custom resize behavior -- input stretches, menu button doesn't
129  inline int inp_x() { return(fltk3::box_dx(box())); }
130  inline int inp_y() { return(fltk3::box_dy(box())); }
131  inline int inp_w() { return(w() - fltk3::box_dw(box()) - 20); }
132  inline int inp_h() { return(h() - fltk3::box_dh(box())); }
133 
134  inline int menu_x() { return(w() - 20 - fltk3::box_dx(box())); }
135  inline int menu_y() { return(fltk3::box_dy(box())); }
136  inline int menu_w() { return(20); }
137  inline int menu_h() { return(h() - fltk3::box_dh(box())); }
