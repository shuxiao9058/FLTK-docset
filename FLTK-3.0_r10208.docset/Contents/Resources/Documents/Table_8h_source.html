227  class FLTK3_EXPORT IntVector {
228  int *arr;
229  unsigned int _size;
230  void init() {
231  arr = NULL;
232  _size = 0;
233  }
234  void copy(int *newarr, unsigned int newsize) {
235  size(newsize);
236  memcpy(arr, newarr, newsize * sizeof(int));
237  }
238  public:
239  IntVector() { init(); } // CTOR
240  ~IntVector() { if ( arr ) free(arr); arr = NULL; } // DTOR
241  IntVector(IntVector&o) { init(); copy(o.arr, o._size); } // COPY CTOR
242  IntVector& operator=(IntVector&o) { // ASSIGN
243  init();
244  copy(o.arr, o._size);
245  return(*this);
246  }
247  int operator[](int x) const { return(arr[x]); }
248  int& operator[](int x) { return(arr[x]); }
249  unsigned int size() { return(_size); }
250  void size(unsigned int count) {
251  if ( count != _size ) {
252  arr = (int*)realloc(arr, count * sizeof(int));
253  _size = count;
254  }
255  }
256  int pop_back() { int tmp = arr[_size-1]; _size--; return(tmp); }
257  void push_back(int val) { unsigned int x = _size; size(_size+1); arr[x] = val; }
258  int back() { return(arr[_size-1]); }
259  };

459  void redraw_range(int toprow, int botrow, int leftcol, int rightcol) {
460  if ( _redraw_toprow == -1 ) {
461  // Initialize redraw range
462  _redraw_toprow = toprow;
463  _redraw_botrow = botrow;
464  _redraw_leftcol = leftcol;
465  _redraw_rightcol = rightcol;
466  } else {
467  // Extend redraw range
468  if ( toprow < _redraw_toprow ) _redraw_toprow = toprow;
469  if ( botrow > _redraw_botrow ) _redraw_botrow = botrow;
470  if ( leftcol < _redraw_leftcol ) _redraw_leftcol = leftcol;
471  if ( rightcol > _redraw_rightcol ) _redraw_rightcol = rightcol;
472  }
473 
474  // Indicate partial redraw needed of some cells
475  damage(fltk3::DAMAGE_CHILD);
476  }
