103  const size_t MAX_FL_HELP_FS_ELTS = 100;
104 
105  struct FLTK3_EXPORT HelpFontStack {
107  HelpFontStack() {
108  nfonts_ = 0;
109  }
110 
111  void init(fltk3::Font f, fltk3::Fontsize s, fltk3::Color c) {
112  nfonts_ = 0;
113  elts_[nfonts_].set(f, s, c);
114  fltk3::font(f, s);
115  fltk3::color(c);
116  }
118  void top(fltk3::Font &f, fltk3::Fontsize &s, fltk3::Color &c) { elts_[nfonts_].get(f, s, c); }
120  void push(fltk3::Font f, fltk3::Fontsize s, fltk3::Color c) {
121  if (nfonts_ < MAX_FL_HELP_FS_ELTS-1) nfonts_ ++;
122  elts_[nfonts_].set(f, s, c);
123  fltk3::font(f, s); fltk3::color(c);
124  }
126  void pop(fltk3::Font &f, fltk3::Fontsize &s, fltk3::Color &c) {
127  if (nfonts_ > 0) nfonts_ --;
128  top(f, s, c);
129  fltk3::font(f, s); fltk3::color(c);
130  }
132  size_t count() const {return nfonts_;} // Gets the current number of fonts in the stack
133 
134  protected:
135  size_t nfonts_;
136  fltk3::HelpFontStyle elts_[100];
137  };
