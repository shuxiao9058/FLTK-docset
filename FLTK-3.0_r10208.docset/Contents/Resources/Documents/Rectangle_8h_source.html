31 // rectangle macros that help keeping rectangle predicates as strict as possible
32 // even when not using rectangles in some situations (as when only using w h scalars)
33 // so that there is only one strict defintion for common predicates,
34 // if one change the following, it will be repercuted in all the core lib
35 #define FLTK3_RECT_EMPTY(w,h) (w <= 0 || h <= 0)
36 // we should always use the same evaluation for center_x, center_y in all corelib code:
37 //#define FLTK_CENTER_X(coord, length) (coord + (length>>1))
38 //#define FLTK_CENTER_Y(coord, length) (coord + (length>>1))
